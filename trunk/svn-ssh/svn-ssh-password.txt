# Save SVN SSH Password

mkdir -p ~/.ssh
chmod 755 ~/.ssh
cd ~/.ssh
ssh-keygen -b 1024 -t dsa

cat ~/.ssh/id_dsa.pub >> ~/.ssh/authorized_keys
